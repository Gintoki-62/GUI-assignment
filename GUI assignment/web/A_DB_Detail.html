CREATE TABLE BOOK ( BOOK_ID VARCHAR(10) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, AUTHOR_NAME VARCHAR(100) NOT NULL, PUBLISHER VARCHAR(100) NOT NULL, NO_OF_PAGES INTEGER, BOOK_DESC VARCHAR(10000) NOT NULL, BOOK_QUANTITY INTEGER, BOOK_TYPE VARCHAR(100), BOOK_IMAGE VARCHAR(1000), BOOK_CATEGORY VARCHAR(100), PRIMARY KEY (BOOK_ID)); CREATE TABLE CART ( CART_ID VARCHAR(10) NOT NULL, BOOK_ID VARCHAR(10) NOT NULL, USER_ID VARCHAR(100) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, BOOK_IMAGE VARCHAR(1000) NOT NULL, QUANTITY INTEGER NOT NULL, PRIMARY KEY (CART_ID, BOOK_ID)); CREATE TABLE PAYMENT ( TRANSACTION_ID VARCHAR(50) NOT NULL, ORDER_ID VARCHAR(50), USER_ID VARCHAR(100), "METHOD" VARCHAR(50), BANK VARCHAR(50) , WALLET VARCHAR(50), AMOUNT DECIMAL(10, 2), PRIMARY KEY (TRANSACTION_ID), FOREIGN KEY (ORDER_ID) REFERENCES ORDERS(ORDER_ID)); CREATE TABLE ORDERS ( ORDER_ID VARCHAR(50) PRIMARY KEY, USER_ID VARCHAR(100) NOT NULL, ORDER_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP, TOTAL_AMOUNT DECIMAL(10, 2) NOT NULL, STATUS VARCHAR(20) NOT NULL DEFAULT 'Pending' ); CREATE TABLE ORDER_ITEMS (ORDER_ITEM_ID VARCHAR(50) NOT NULL, ORDER_ID VARCHAR(50) NOT NULL, BOOK_ID VARCHAR(10) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, BOOK_IMAGE VARCHAR(1000) NOT NULL, QUANTITY INTEGER NOT NULL, PRIMARY KEY (ORDER_ITEM_ID)); CREATE TABLE ACCOUNT ( PROFILE VARCHAR(255), USERID VARCHAR(10) NOT NULL, USERNAME VARCHAR(50) NOT NULL, EMAIL VARCHAR(50), PASSWORD VARCHAR(50), GENDER CHAR(1), PRIMARY KEY (USERID, USERNAME)); CREATE TABLE REGISTER ( ID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1 NOT NULL GENERATED ALWAYS AS IDENTITY, "NAME" VARCHAR(100), PHONE VARCHAR(15), EMAIL VARCHAR(100), USERNAME VARCHAR(50), PASSWORD VARCHAR(100), GENDER VARCHAR(10), ADDRESS VARCHAR(255), PRIMARY KEY (ID));
