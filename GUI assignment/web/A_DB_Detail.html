CREATE TABLE BOOK ( BOOK_ID VARCHAR(10) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, AUTHOR_NAME VARCHAR(100) NOT NULL, PUBLISHER VARCHAR(100) NOT NULL, NO_OF_PAGES INTEGER, BOOK_DESC VARCHAR(10000) NOT NULL, BOOK_QUANTITY INTEGER, BOOK_TYPE VARCHAR(100), BOOK_IMAGE VARCHAR(1000), BOOK_CATEGORY VARCHAR(100), PRIMARY KEY (BOOK_ID)); CREATE TABLE CART ( CART_ID VARCHAR(10) NOT NULL, BOOK_ID VARCHAR(10) NOT NULL, USER_ID VARCHAR(10) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, BOOK_IMAGE VARCHAR(1000) NOT NULL, QUANTITY INTEGER NOT NULL, PRIMARY KEY (CART_ID, BOOK_ID)); CREATE TABLE PAYMENT ( TRANSACTION_ID VARCHAR(50) NOT NULL, ORDER_ID VARCHAR(50), USER_ID VARCHAR(50), "METHOD" VARCHAR(50), BANK VARCHAR(50), WALLET VARCHAR(50), AMOUNT DECIMAL(10, 2), CART_ID VARCHAR(10), PRIMARY KEY (TRANSACTION_ID)); CREATE TABLE ACCOUNT ( PROFILE VARCHAR(255), USERID VARCHAR(10) NOT NULL, USERNAME VARCHAR(50) NOT NULL, EMAIL VARCHAR(50), PASSWORD VARCHAR(50), GENDER CHAR(1), PRIMARY KEY (USERID, USERNAME));
