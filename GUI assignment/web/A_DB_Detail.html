CREATE TABLE BOOK ( BOOK_ID VARCHAR(10) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, AUTHOR_NAME VARCHAR(100) NOT NULL, PUBLISHER VARCHAR(100) NOT NULL, NO_OF_PAGES INTEGER, BOOK_DESC VARCHAR(10000) NOT NULL, BOOK_QUANTITY INTEGER, BOOK_TYPE VARCHAR(100), BOOK_IMAGE VARCHAR(1000), BOOK_CATEGORY VARCHAR(100) PRIMARY KEY (BOOK_ID)); CREATE TABLE CART ( CART_ID VARCHAR(10) NOT NULL, BOOK_ID VARCHAR(10) NOT NULL, USER_ID VARCHAR(10) NOT NULL, BOOK_NAME VARCHAR(100) NOT NULL, BOOK_PRICE DECIMAL(10, 2) NOT NULL, BOOK_IMAGE VARCHAR(1000) NOT NULL, QUANTITY INTEGER NOT NULL, PRIMARY KEY (CART_ID, BOOK_ID));
